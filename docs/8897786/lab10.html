Feature extraction with a pretrained model

We will use the VGG16 model, which is a model trained on the ImageNet dataset, we need to specify some parameters to have a successful feature extraction, we will use the following parameters:

weights: specifies the weight checkpoint from which to initialize the model.

include_top: refers to including (or not) the densely connected classifier on top of the network. By default, this densely connected classifier corresponds to the 1000 classes from ImageNet. Since we intend to use our own densely connected classifier (with only two classes: cat and dog), we don't need to include it.

import tensorflow as tf
from keras.applications.vgg16 import VGG16
from keras.layers import Flatten, Dense, Dropout

# Loading the VGG16 model without the top layers
conv_base = VGG16(weights='imagenet', include_top=False, input_shape=(224, 224, 3))

# Freezing the layers of the pre-trained VGG16 model
for layer in conv_base.layers:
    layer.trainable = False

# Unfreezzed the last four layers of the VGG16 model
for layer in conv_base.layers[:-4]:
    layer.trainable = False

# Creating the custom model for fine-tuning
inputs = tf.keras.Input(shape=(224, 224, 3))
x = conv_base(inputs, training=False)
x = Flatten()(x) # Flatten the output of the VGG16 base
x = Dense(256, activation='relu')(x) # Adding a dense layer with ReLU activation
x = Dropout(0.5)(x) # Add dropout for regularization
outputs = Dense(3, activation='softmax')(x) # Adding a dense output layer with softmax activation

model = tf.keras.Model(inputs, outputs)

# Compiling the model
model.compile(optimizer='adam', loss='categorical_crossentropy', metrics=['accuracy'])

# Fine-tuning the model
num_epochs = 10
history = model.fit(
    train_dataset,
    epochs=num_epochs,
    validation_data=validation_dataset
)

# Binarizing the true labels
true_labels_bin = label_binarize(test_dataset.classes, classes=range(len(top_classes)))

# Computing ROC curve and AUC for each class
fpr = dict()
tpr = dict()
roc_auc = dict()

for i in range(len(top_classes)):
    fpr[i], tpr[i], _ = roc_curve(true_labels_bin[:, i], predictions[:, i])
    roc_auc[i] = auc(fpr[i], tpr[i])

# Plotting ROC curve for each class
plt.figure(figsize=(10, 6))
for i in range(len(top_classes)):
    plt.plot(fpr[i], tpr[i], label=f'Class {i} (AUC = {roc_auc[i]:.2f})')
